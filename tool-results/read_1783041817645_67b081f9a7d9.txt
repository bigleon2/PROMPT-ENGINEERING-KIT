   334→   334→  if (session.results) {
   335→   335→    message += '## Résultats attendus\n\n';
   336→   336→    message += session.results + '\n\n';
   337→   337→  }
   338→   338→
   339→   339→  if (session.constraints) {
   340→   340→    message += '## Contraintes additionnelles\n\n';
   341→   341→    message += session.constraints + '\n\n';
   342→   342→  }
   343→   343→
   344→   344→  if (previousOutputs.length > 0) {
   345→   345→    message += '## Résultats des étapes précédentes\n\n';
   346→   346→    for (const prev of previousOutputs) {
   347→   347→      const stepName = allInstructions[prev.stepNumber - 1]?.name || 'Inconnue';
   348→   348→      message += '### Étape ' + prev.stepNumber + ' — ' + stepName + '\n\n';
   349→   349→      message += prev.output + '\n\n';
   350→   350→    }
   351→   351→    message += '---\n\n';
   352→   352→  }
   353→   353→
   354→   354→  message += step.instruction;
   355→   355→  return message;
   356→   356→}

   358→   358→export async function POST(
   359→   359→  request: NextRequest,
   360→   360→  { params }: { params: Promise<{ id: string }> }
   361→   361→) {
   362→   362→  try {
   363→   363→    const { id } = await params;
   364→   364→    const zai = await ZAI.create();
   365→   365→
   366→   366→    // Load session with steps
   367→   367→    const session = await db.session.findUnique({
   368→   368→      where: { id },
   369→   369→      include: {
   370→   370→        steps: {
   371→   371→          orderBy: { stepNumber: 'asc' },
   372→   372→        },
   373→   373→      },
   374→   374→    });
   375→   375→
   376→   376→    if (!session) {
   377→   377→      return NextResponse.json({ success: false, error: 'Session non trouvée' }, { status: 404 });
   378→   378→    }
   379→   379→
   380→   380→    // Determine execution mode and instructions
   381→   381→    const executionMode = session.executionMode || 'cot';
   382→   382→    const isChaining = executionMode === 'chaining' || (executionMode === 'hybride' && session.complexity === 'simple');
   383→   383→    const instructions = isChaining ? CHAINING_STEP_INSTRUCTIONS : STEP_INSTRUCTIONS;
   384→   384→
   385→   385→    // Build the PEK system prompt
   386→   386→    const systemPrompt = buildSystemPrompt(session);
   387→   387→
   388→   388→    const previousOutputs: { stepNumber: number; output: string }[] = [];
   389→   389→    const stepResults: { stepNumber: number; name: string; output: string; duration: number; status: string }[] = [];
   390→   390→
   391→   391→    // Execute each step sequentially
   392→   392→    for (const step of session.steps) {
   393→   393→      const stepStartTime = Date.now();
   394→   394→
   395→   395→      // Mark step as running
   396→   396→      await db.step.update({
   397→   397→        where: { id: step.id },
   398→   398→        data: { status: 'running' },
   399→   399→      });
   400→   400→
   401→   401→      // Update session current step
   402→   402→      await db.session.update({
   403→   403→        where: { id },
   404→   404→        data: { currentStep: step.stepNumber },
   405→   405→      });
   406→   406→
   407→   407→      const stepConfig = instructions[step.stepNumber - 1];
   408→   408→      const userMessage = buildUserMessage(stepConfig, session, previousOutputs, instructions);
   409→   409→
   410→   410→      try {
   411→   411→        const completion = await zai.chat.completions.create({
   412→   412→          messages: [
   413→   413→            { role: 'assistant', content: systemPrompt },
   414→   414→            { role: 'user', content: userMessage },
   415→   415→          ],
   416→   416→          thinking: { type: 'disabled' },
   417→   417→        });
   418→   418→
   419→   419→        const output = completion.choices[0]?.message?.content || 'Erreur: aucune réponse générée';
   420→   420→        const duration = Math.round((Date.now() - stepStartTime) / 1000);
   421→   421→
   422→   422→        // Save step output
   423→   423→        await db.step.update({
   424→   424→          where: { id: step.id },
   425→   425→          data: {
   426→   426→            status: 'completed',
   427→   427→            output,
   428→   428→            duration,
   429→   429→          },
   430→   430→        });
   431→   431→
   432→   432→        previousOutputs.push({ stepNumber: step.stepNumber, output });
   433→   433→        stepResults.push({
   434→   434→          stepNumber: step.stepNumber,
   435→   435→          name: step.name,
   436→   436→          output,
   437→   437→          duration,
   438→   438→          status: 'completed',
   439→   439→        });
   440→   440→
   441→   441→        // After validation step, compute and save validation score
   442→   442→        // CoT: step 6, Chaining: step 3
   443→   443→        const validationStepNumber = isChaining ? 3 : 6;
   444→   444→        if (step.stepNumber === validationStepNumber) {
   445→   445→          const validationResult = await computeAndSaveValidation(
   446→   446→            id,
   447→   447→            session.selectedBlocks,
   448→   448→            output,
   449→   449→            previousOutputs,
   450→   450→            zai,
   451→   451→            systemPrompt,
   452→   452→            { complexity: session.complexity, language: session.language, constraints: session.constraints },
   453→   453→          );
   454→   454→
   455→   455→          // PHASE 2: Auto-iteration if score < 22/25 (max 1 retry)
   456→   456→          if (!validationResult.passed && validationResult.totalScore < 22) {
   457→   457→            // Reset the improvement/amélioration step for retry
   458→   458→            const improvementStep = session.steps.find(
   459→   459→              (s) => s.stepNumber === (isChaining ? 4 : 7)
   460→   460→            );
   461→   461→            if (improvementStep) {
   462→   462→              await db.step.update({
   463→   463→                where: { id: improvementStep.id },
   464→   464→                data: {
   465→   465→                  status: 'pending',
   466→   466→                  output: '',
   467→   467→                  duration: 0,
   468→   468→                },
   469→   469→              });
   470→   470→
   471→   471→              // Add feedback from validation to previous outputs
   472→   472→              previousOutputs.push({
   473→   473→                stepNumber: 100,
   474→   474→                output: `FEEDBACK DE VALIDATION (score ${validationResult.totalScore}/25 — seuil non atteint) :\n` +
   475→   475→                  `- Complétude: ${validationResult.completeness}/10\n` +
   476→   476→                  `- Cohérence: ${validationResult.coherence}/8\n` +
   477→   477→                  `- Qualité: ${validationResult.quality}/7\n\n` +
   478→   478→                  `Corrige les déficiences identifiées et régénère le kit amélioré.`,
   479→   479→              });
   480→   480→            }
   481→   481→          }
   482→   482→        }
   483→   483→      } catch (llmError) {
   484→   484→        const duration = Math.round((Date.now() - stepStartTime) / 1000);
   485→   485→        const errorMsg = llmError instanceof Error ? llmError.message : 'Erreur LLM inconnue';
   486→   486→
   487→   487→        await db.step.update({
   488→   488→          where: { id: step.id },
   489→   489→          data: {
   490→   490→            status: 'error',
   491→   491→            output: 'Erreur: ' + errorMsg,
   492→   492→            duration,
   493→   493→          },
   494→   494→        });
   495→   495→
   496→   496→        stepResults.push({
   497→   497→          stepNumber: step.stepNumber,
   498→   498→          name: step.name,
   499→   499→          output: 'Erreur: ' + errorMsg,
   500→   500→          duration,
   501→   501→          status: 'error',
   502→   502→        });
   503→   503→
   504→   504→        // Mark remaining steps as skipped
   505→   505→        for (const remainingStep of session.steps) {
   506→   506→          if (remainingStep.stepNumber > step.stepNumber && remainingStep.status === 'pending') {
   507→   507→            await db.step.update({
   508→   508→              where: { id: remainingStep.id },
   509→   509→              data: { status: 'skipped' },
   510→   510→            });
   511→   511→          }
   512→   512→        }
   513→   513→
   514→   514→        // Update session status
   515→   515→        await db.session.update({
   516→   516→          where: { id },
   517→   517→          data: { status: 'error' },
   518→   518→        });
   519→   519→
   520→   520→        return NextResponse.json({
   521→   521→          success: false,
   522→   522→          error: 'Erreur à l\'étape ' + step.stepNumber + ': ' + errorMsg,
   523→   523→          data: { steps: stepResults },
   524→   524→        }, { status: 500 });
   525→   525→      }
   526→   526→    }
   527→   527→
   528→   528→    // All steps completed successfully
   529→   529→    await db.session.update({
   530→   530→      where: { id },
   531→   531→      data: { status: 'completed', currentStep: 8 },
   532→   532→    });
   533→   533→
   534→   534→    return NextResponse.json({
   535→   535→      success: true,
   536→   536→      data: {
   537→   537→        sessionId: id,
   538→   538→        steps: stepResults,
   539→   539→      },
   540→   540→    });
   541→   541→  } catch (error) {
   542→   542→    const message = error instanceof Error ? error.message : 'Erreur lors de l\'exécution du pipeline';
   543→   543→    return NextResponse.json({ success: false, error: message }, { status: 500 });
   544→   544→  }
   545→   545→}

   547→   547→async function computeAndSaveValidation(
   548→   548→  sessionId: string,
   549→   549→  selectedBlocks: string,
   550→   550→  validationOutput: string,
   551→   551→  allStepOutputs: { stepNumber: number; output: string }[],
   552→   552→  zaiInstance: Awaited<ReturnType<typeof ZAI.create>>,
   553→   553→  systemPrompt: string,
   554→   554→  sessionContext: { complexity: string; language: string; constraints: string },
   555→   555→): Promise<{ completeness: number; coherence: number; quality: number; totalScore: number; maxScore: number; passed: boolean }> {
   556→   556→  const blocks = selectedBlocks === 'auto'
   557→   557→    ? ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'I', 'J']
   558→   558→    : selectedBlocks.split(',').map((b) => b.trim().toUpperCase()).filter(Boolean);
   559→   559→
   560→   560→  // --- Heuristic scoring (fast, always runs) ---
   561→   561→  let blocksFound = 0;
   562→   562→  const combinedOutput = validationOutput + '\n' + (allStepOutputs.find((s) => s.stepNumber === 5)?.output || '');
   563→   563→  const lowerCombined = combinedOutput.toLowerCase();
   564→   564→  for (const block of blocks) {
   565→   565→    const patterns = ['Bloc ' + block, 'bloc ' + block, block + ' :', block + ':'];
   566→   566→    if (patterns.some((p) => combinedOutput.includes(p))) {
   567→   567→      blocksFound++;
   568→   568→    }
   569→   569→  }
   570→   570→  const heuristicCompleteness = blocks.length > 0 ? Math.min(10, Math.round((blocksFound / blocks.length) * 10)) : 0;
   571→   571→
   572→   572→  const domainKeywords = ['web', 'data', 'iot', 'éducation', 'education', 'marketing', 'recherche', 'finance', 'santé', 'health'];
   573→   573→  const domainCount = domainKeywords.filter((d) => lowerCombined.includes(d)).length;
   574→   574→  const has4PlusDomains = domainCount >= 4;
   575→   575→  const hasPlaceholders = lowerCombined.includes('à compléter') || lowerCombined.includes('placeholder');
   576→   576→  const hasErrorHandling = lowerCombined.includes('erreur') || lowerCombined.includes('error') || lowerCombined.includes('cas limite');
   577→   577→  const hasFeedback = lowerCombined.includes('feedback') || lowerCombined.includes('retour') || lowerCombined.includes('itération');
   578→   578→  const hasFormatSpec = lowerCombined.includes('format');
   579→   579→  const hasInstructions = lowerCombined.includes('instruction') || lowerCombined.includes('guide');
   580→   580→
   581→   581→  // --- LLM-based scoring (accurate, used as primary) ---
   582→   582→  let llmCompleteness = 0;
   583→   583→  let llmCoherence = 0;
   584→   584→  let llmQuality = 0;
   585→   585→  let llmChecks: { label: string; passed: boolean; detail: string }[] = [];
   586→   586→
   587→   587→  try {
   588→   588→    const allOutputsText = allStepOutputs
   589→   589→      .map((s) => '--- ÉTAPE ' + s.stepNumber + ' ---\n' + s.output.slice(0, 3000))
   590→   590→      .join('\n\n');
   591→   591→
   592→   592→    const scoringPrompt = `Tu es un évaluateur strict de kits de prompt engineering. Analyse le contenu ci-dessous et attribue un score.
   593→   593→
   594→   594→BLOCS ATTENDUS : ${blocks.join(', ')}
   595→   595→COMPLEXITÉ : ${sessionContext.complexity}
   596→   596→LANGUE : ${sessionContext.language}
   597→   597→
   598→   598→CONTENU À ÉVALUER :
   599→   599→${allOutputsText.slice(0, 8000)}
   600→   600→
   601→   601→Réponds UNIQUEMENT en JSON valide (pas de markdown, pas de backticks) avec cette structure exacte :
   602→   602→{"completeness": <0-10>, "coherence": <0-8>, "quality": <0-7>, "checks": [{"label": "<description>", "passed": <true|false>, "detail": "<raison>"}]}
   603→   603→
   604→   604→Critères :
   605→   605→- completeness (10pts) : Tous les blocs attendus sont présents et non vides
   606→   606→- coherence (8pts) : Terminologie uniforme, pas de contradictions, références croisées valides
   607→   607→- quality (7pts) : Langage clair, exemples concrets dans 4+ domaines, Markdown correct
   608→   608→
   609→   609→Checks obligatoires (12 items) :
   610→   610→1. Complétude des blocs (tous présents)
   611→   611→2. Références croisées cohérentes
   612→   612→3. Exemples dans 4+ domaines variés
   613→   613→4. Préservation du contenu original
   614→   614→5. Fidélité au projet source
   615→   615→6. Structuration conforme PEK (blocs A-J)
   616→   616→7. Variables et placeholders définis
   617→   617→8. Cas d'erreur et limites couverts
   618→   618→9. Instructions claires et non ambigües
   619→   619→10. Formats de sortie spécifiés
   620→   620→11. Pas de greenwashing ni fausses promesses
   621→   621→12. Mécanismes de feedback et itération prévus`;
   622→   622→
   623→   623→    const completion = await zaiInstance.chat.completions.create({
   624→   624→      messages: [
   625→   625→        { role: 'assistant', content: systemPrompt },
   626→   626→        { role: 'user', content: scoringPrompt },
   627→   627→      ],
   628→   628→      thinking: { type: 'disabled' },
   629→   629→    });
   630→   630→
   631→   631→    const raw = completion.choices[0]?.message?.content || '';
   632→   632→    const jsonMatch = raw.match(/\{[\s\S]*\}/);
   633→   633→    if (jsonMatch) {
   634→   634→      const parsed = JSON.parse(jsonMatch[0]);
   635→   635→      llmCompleteness = Math.max(0, Math.min(10, Math.round(parsed.completeness || 0)));
   636→   636→      llmCoherence = Math.max(0, Math.min(8, Math.round(parsed.coherence || 0)));
   637→   637→      llmQuality = Math.max(0, Math.min(7, Math.round(parsed.quality || 0)));
   638→   638→      if (Array.isArray(parsed.checks) && parsed.checks.length >= 10) {
   639→   639→        llmChecks = parsed.checks.slice(0, 12).map((c: { label: string; passed: boolean; detail?: string }) => ({
   640→   640→          label: String(c.label),
   641→   641→          passed: Boolean(c.passed),
   642→   642→          detail: String(c.detail || ''),
   643→   643→        }));
   644→   644→      }
   645→   645→    }
   646→   646→  } catch {
   647→   647→    // LLM scoring failed — fall back to heuristic
   648→   648→  }
   649→   649→
   650→   650→  // Use LLM scores if available, otherwise heuristic
   651→   651→  const completeness = llmCompleteness > 0 ? llmCompleteness : heuristicCompleteness;
   652→   652→  const coherence = llmCoherence > 0 ? llmCoherence : 8;
   653→   653→  const quality = llmQuality > 0 ? llmQuality : (has4PlusDomains ? 7 : hasErrorHandling ? 6 : 5);
   654→   654→  const totalScore = completeness + coherence + quality;
   655→   655→  const maxScore = 25;
   656→   656→
   657→   657→  const checks = llmChecks.length >= 10
   658→   658→    ? llmChecks
   659→   659→    : [
   660→   660→        { label: 'Complétude des blocs', passed: completeness >= 7, detail: blocksFound + '/' + blocks.length + ' blocs trouvés' },
   661→   661→        { label: 'Références croisées cohérentes', passed: coherence >= 6, detail: 'Cohérence ' + coherence + '/8' },
   662→   662→        { label: 'Exemples dans 4+ domaines', passed: has4PlusDomains, detail: domainCount + ' domaines détectés' },
   663→   663→        { label: 'Préservation du contenu original', passed: true, detail: 'Non vérifiable automatiquement' },
   664→   664→        { label: 'Fidélité au projet source', passed: true, detail: 'Non vérifiable automatiquement' },
   665→   665→        { label: 'Structuration conforme PEK', passed: blocksFound > 0, detail: 'Blocs A-J détectés' },
   666→   666→        { label: 'Variables et placeholders définis', passed: hasPlaceholders, detail: hasPlaceholders ? 'Trouvés' : 'Non trouvés' },
   667→   667→        { label: "Cas d'erreur couverts", passed: hasErrorHandling, detail: hasErrorHandling ? 'Trouvés' : 'Non trouvés' },
   668→   668→        { label: 'Instructions claires', passed: hasInstructions, detail: hasInstructions ? 'Trouvées' : 'Non trouvées' },
   669→   669→        { label: 'Formats de sortie spécifiés', passed: hasFormatSpec, detail: hasFormatSpec ? 'Trouvés' : 'Non trouvés' },
   670→   670→        { label: 'Pas de greenwashing', passed: true, detail: 'Non vérifiable automatiquement' },
   671→   671→        { label: 'Feedback et itération prévus', passed: hasFeedback, detail: hasFeedback ? 'Trouvés' : 'Non trouvés' },
   672→   672→      ];
   673→   673→
   674→   674→  const passed = totalScore >= 22;
   675→   675→
   676→   676→  await db.validation.create({
   677→   677→    data: {
   678→   678→      sessionId,
   679→   679→      completeness,
   680→   680→      coherence,
   681→   681→      quality,
   682→   682→      totalScore,
   683→   683→      maxScore,
   684→   684→      checks: JSON.stringify(checks),
   685→   685→      passed,
   686→   686→    },
   687→   687→  });
   688→   688→
   689→   689→  return { completeness, coherence, quality, totalScore, maxScore, passed };
   690→   690→}
